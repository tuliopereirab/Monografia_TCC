\begin{center}
\includegraphics[scale=1]{apendice/images/diag_casos.jpeg}
\end{center}

\begin{description}
	\item[Realizar simulações com \textit{Quartus}]: \textit{pyConv} gera o arquivo de inicialização de memória para essa ferramenta (\textit{.mif});
	\item[Realizar simulações com \textit{Modelsim}]: \textit{pyConv} gera o arquivo de inicialização de memória para essa ferramenta (\textit{.mem}); 
	\item[Executar com FPGA]: \textit{pyConv} gera o arquivo de inicialização de memória \textit{.mif} após realizar simulações para evitar erros;
	\item[Verificar se algoritmo funcionará no \hw{}]: \textit{pyConv} realiza uma simulação para verificar se o algoritmo inserido está livre de problemas que leve a arquitetura a um estado de erro;
	\item[Verificar algoritmo criado sem ler código binário]: \textit{pyConv} gera um arquivo de texto com as instruções e argumentos escritos conforme documentação da linguagem \py{}.
\end{description}
